设计模式策略模式-不解释

定义封装独立算法族,可以相互替换



/*
 *策略模式(定义封装独立算法族,可以相互替换)
 *定义不同算法
 *不同的英雄使用不同的武器发起战争和逃跑
 * 武器和逃跑可以理解为算法,基于组合实现可以动态修改英雄行为
 */
public class StrategyPattern {


    public static void main(String args[]){
        Hero hero  = new   IronMan();
        hero.heroBehaviour();
        //钢铁侠更换武器和逃跑套路(更换算法)
        //更换逃跑行为
        hero.setEscape(new UndergroundEscape());
        //更好战争行为
        hero.setWar(new HammerWar());
        hero.heroBehaviour();
    }


}

/**
 * 战争
 * */
interface War{
    void war();
}

/**
 * 逃跑
 * */
interface Escape{
    void escape();
}

class HammerWar implements War{

    @Override
    public void war() {
     System.out.println(" 铁锤");
    }
}

class MissileWar implements  War{
    @Override
    public void war() {
        System.out.println("导弹");
    }
}


class NoWar implements  War{

    @Override
    public void war() {
        System.out.println("我不是超人,我不会发起战争");
    }
}

/**
 * 飞走逃跑
 * */
class FlyEscape implements Escape{

    @Override
    public void escape() {
        System.out.println("飞走");
    }
}


/**
 * 地下逃跑
 * */
class UndergroundEscape implements Escape{

    @Override
    public void escape() {
        System.out.println("地下");
    }
}


class NoEscape implements  Escape{

    @Override
    public void escape() {
        System.out.println("我是不会逃跑的");
    }
}



/**
 *超类==> 超人
 * */
abstract class Hero{

    War war;
    Escape escape;


    public void war(){
        war.war();
    }

    public void escape(){
        escape.escape();
    }

    public void setWar(War war) {
        this.war = war;
    }

    public void setEscape(Escape escape) {
        this.escape = escape;
    }

    //英雄行为
    abstract void heroBehaviour();

}

/**
 * 钢铁侠
 * */
class IronMan extends Hero{

    public IronMan() {
        war = new MissileWar();
        escape = new  FlyEscape();
    }

    @Override
    void heroBehaviour() {
        System.out.println("我是钢铁侠");
        super.war();
        super.escape();

    }
}


/**
 * 普通人
 * */
class OrdinaryMan extends Hero{

    public OrdinaryMan() {
        war = new NoWar();
        escape = new NoEscape();
    }

    @Override
    void heroBehaviour() {
        System.out.println("我是普通人");
        super.war();
        super.escape();
    }
}
